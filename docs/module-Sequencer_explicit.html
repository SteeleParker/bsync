//Return the sequencer for use elsewhere
//Implied args
var bsync = require('bsync')

var sequence1 = bsync.sequence([
    function1,
    function2
],{
    arg1 : "String",

},function(err, data) {
    //Do something with the result of the final output
});
//Implied Args
bsync.sequence([
    function1,
    function2
],{
    arg1 : "String",
    arg2 : true
},function(err, data) {
    //Do something with the result of the final output
});
//Explicit args
bsync.sequence({
    sequence:[
        function1,
        function2
    ],
    args: {
        arg1 : "String",
        arg2 : true
    },
    callback: function(err, data) {
        //Do something with the result of the final output
    }
});
